    <script>
        let currentLang = 'cz';

        // 1. Language Toggle Logic
        function setLanguage(lang) {
            currentLang = lang;
            const elements = document.querySelectorAll('[data-cz]');
            
            elements.forEach(el => {
                el.textContent = el.getAttribute(`data-${lang}`);
            });

            // Update Toggle Button Styles
            document.getElementById('btn-cz').classList.toggle('lang-active', lang === 'cz');
            document.getElementById('btn-en').classList.toggle('lang-active', lang === 'en');
            
            // Mobile label
            document.getElementById('mobile-lang-label').textContent = lang === 'cz' ? 'EN' : 'CZ';
            
            // Set HTML lang attribute for accessibility
            document.documentElement.lang = lang;
        }

        // 2. Menu Filter Logic
        function filterMenu(category) {
            const container = document.getElementById('menu-container');
            const btns = document.querySelectorAll('.menu-tab-btn');
            
            // Update buttons
            btns.forEach(btn => {
                btn.classList.remove('bg-espresso', 'text-white');
                btn.classList.add('border', 'border-espresso/20');
            });
            event.target.classList.add('bg-espresso', 'text-white');
            event.target.classList.remove('border', 'border-espresso/20');

            // Toggle Content (This is a simplified version for the demo)
            // In a full app, you'd hide/show specific item categories.
            const categories = ['espresso', 'sweet', 'savory'];
            
            // This demo logic swaps entire HTML blocks for speed
            if(category === 'espresso') {
                container.innerHTML = `
                    <div class="flex justify-between items-end border-b border-espresso/10 pb-2 group">
                        <div><h3 class="text-xl font-semibold group-hover:text-accent">Espresso</h3><p class="text-sm opacity-60 italic">Classic</p></div>
                        <span class="text-xl font-serif">55 Kč</span>
                    </div>
                    <div class="flex justify-between items-end border-b border-espresso/10 pb-2 group">
                        <div><h3 class="text-xl font-semibold group-hover:text-accent">Cappuccino</h3><p class="text-sm opacity-60 italic">Creamy</p></div>
                        <span class="text-xl font-serif">75 Kč</span>
                    </div>
                `;
            } else if(category === 'sweet') {
                container.innerHTML = `
                    <div class="flex justify-between items-end border-b border-espresso/10 pb-2 group">
                        <div><h3 class="text-xl font-semibold group-hover:text-accent">Croissant</h3><p class="text-sm opacity-60 italic">Fresh daily</p></div>
                        <span class="text-xl font-serif">45 Kč</span>
                    </div>
                    <div class="flex justify-between items-end border-b border-espresso/10 pb-2 group">
                        <div><h3 class="text-xl font-semibold group-hover:text-accent">Cheesecake</h3><p class="text-sm opacity-60 italic">New York style</p></div>
                        <span class="text-xl font-serif">95 Kč</span>
                    </div>
                `;
            } else {
                container.innerHTML = `
                    <div class="flex justify-between items-end border-b border-espresso/10 pb-2 group">
                        <div><h3 class="text-xl font-semibold group-hover:text-accent">Avocado Toast</h3><p class="text-sm opacity-60 italic">With poached egg</p></div>
                        <span class="text-xl font-serif">165 Kč</span>
                    </div>
                `;
            }
        }

        // 3. Mobile Menu Toggle
        const menuBtn = document.getElementById('mobile-menu-btn');
        const mobileMenu = document.getElementById('mobile-menu');

        menuBtn.addEventListener('click', () => {
            mobileMenu.classList.toggle('hidden');
        });

        // Close menu on click link
        document.querySelectorAll('#mobile-menu a').forEach(link => {
            link.addEventListener('click', () => mobileMenu.classList.add('hidden'));
        });

        // 4. Sticky Nav Effect
        window.addEventListener('scroll', () => {
            const nav = document.querySelector('nav');
            if (window.scrollY > 50) {
                nav.classList.add('py-2', 'shadow-md');
                nav.classList.remove('py-0');
            } else {
                nav.classList.remove('py-2', 'shadow-md');
            }
        });
    </script>
